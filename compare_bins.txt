'''
objdump -d  -j .text ./bin/ls > ./disassemble/ls
objdump -d  -j .text ./bin/ls_obfuscated > ./disassemble/ls_obfuscated
diff_count=$(cmp -l ./disassemble/ls ./disassemble/ls_obfuscated | wc -l)
text_size1=$(stat -c %s ./disassemble/ls)
percentage_diff=$(echo "scale=4; ($diff_count / $text_size1) * 100" | bc)
echo "Porcentagem de diferença da seção .text: $percentage_diff%"
'''
